CREATE SCHEMA DELIVERY; 

CREATE TABLE DELIVERY.TBUSUARIO ( 
    USUCODIGO SERIAL NOT NULL, 
    USUNOME VARCHAR (60) NOT NULL,
    USUCPF VARCHAR (14) NOT NULL, 
    USUEMAIL VARCHAR (100) NOT NULL,
    USUTELEFONE VARCHAR (16) NOT NULL,
    USUSENHA VARCHAR(20) NOT NULL,
    USUDATANASCIMENTO DATE,  
    CONSTRAINT PK_TBUSUARIO 
    PRIMARY KEY (USUCODIGO),
    CONSTRAINT FK_ENDERECOUSUARIO
    FOREIGN KEY (USUCODIGO) 
    REFERENCES DELIVERY.TBUSUARIO(USUCODIGO)  
); 

CREATE TABLE DELIVERY.TBENDERECO ( 
    ENDCODIGO SERIAL NOT NULL,
    USUCODIGO INT NOT NULL, 
    ENDESTADO VARCHAR(60) NOT NULL,
    ENDCIDADE VARCHAR(60) NOT NULL, 
    ENDBAIRRO VARCHAR(60) NOT NULL,
    ENDRUA VARCHAR (100) NOT NULL,
    ENDNUMERO VARCHAR(100) NOT NULL,	
    CONSTRAINT PK_TBENDERECO 
    PRIMARY KEY (ENDCODIGO),
    CONSTRAINT FK_ENDERECOUSUARIO
    FOREIGN KEY (USUCODIGO) 
    REFERENCES DELIVERY.TBUSUARIO(USUCODIGO)  
); 


CREATE TABLE DELIVERY.TBPRODUTO ( 
    PROCODIGO SERIAL NOT NULL,
    PRIMARY KEY (PROCODIGO),
    USUCODIGO INT NOT NULL, 
    CONSTRAINT FK_PRODUTOUSUARIO
    FOREIGN KEY (USUCODIGO) 
    REFERENCES DELIVERY.TBUSUARIO(USUCODIGO),
    PRONOME VARCHAR (60) NOT NULL,
    PROVALOR NUMERIC (10,2) NOT NULL
        
); 

CREATE TABLE DELIVERY.TBPEDIDO ( 
    PEDCODIGO SERIAL NOT NULL, 
    USUCODIGO INT NOT NULL,
    PEDSTATUS VARCHAR(20) NOT NULL,
    PEDVALOR NUMERIC (10,2) NOT NULL,
    PEDDATA DATE NOT NULL,
    PRIMARY KEY (PEDCODIGO),
    CONSTRAINT FK_PEDIDOUSUARIO
    FOREIGN KEY (USUCODIGO) 
    REFERENCES DELIVERY.TBUSUARIO(USUCODIGO)
);
